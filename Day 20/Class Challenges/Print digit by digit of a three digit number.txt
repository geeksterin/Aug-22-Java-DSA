import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        
        System.out.println(n%10);
        n = n/10;
        
        System.out.println(n%10);
        n = n/10;
        
        System.out.println(n%10);
        n = n/10;
    }
}
